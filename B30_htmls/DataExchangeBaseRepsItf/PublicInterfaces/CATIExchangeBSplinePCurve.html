#else extern "C" const IID IID_CATIExchangeBSplinePCurve ; #endif /** * Interface to exchange BSpline PCurves. * <b>Role</b>: This interface is dedicated to get BSpline PCurves information. * @see CATIExchangeBSplineCurve, CATIExchangeBSplineSurface */ class ExportedByDataExchangeBaseRepsItfCPP CATIExchangeBSplinePCurve: public CATBaseUnknown { CATDeclareInterface;
