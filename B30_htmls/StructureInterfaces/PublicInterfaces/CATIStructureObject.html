//------------------------------------------------------------------ class CATListValCATUnicodeString; /** * Interface to manage the structure object information. * <b>Role</b>: To retrieve data from a structure object * such as attributes and materials. */ class CATIStructureObject: public IUnknown { public: /** * Get the list attribute names of a structure object. * @param CATListValCATUnicodeString * Returns the list of attribute names of a structure object. * @return * An HRESULT value. * <br><b>Legal values</b>: * <dl> * <dt>S_OK</dt> * <dd>operation is successful</dd> * <dt>E_FAIL</dt> * <dd>operation failed</dd> * </dl> */ virtual HRESULT ListAttributeName(CATListValCATUnicodeString **oListOfAttributeNames)=0; }; #endif
