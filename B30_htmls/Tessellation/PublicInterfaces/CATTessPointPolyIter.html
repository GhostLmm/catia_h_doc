#ifndef CATTessPointPolyIter_H #define CATTessPointPolyIter_H // COPYRIGHT DASSAULT SYSTEMES 2004 /** * @CAA2Level L1 * @CAA2Usage U1 */ #include "TessAPI.h" #include "CATMathDef.h" #include "CATMathVector.h" #include "CATCGMNewArray.h" #include "CATIAV5Level.h" #include "CATCGMTessPointPolyIter.h" class CATFace; class CATTessPoint; /** * Class defining the iterator on the points resulting from a Tessellation operator. * <br> Such iterator is directly created by the @href CATBodyTessellator#GetFace * or @href CATCellTessellator#GetFace methods.
