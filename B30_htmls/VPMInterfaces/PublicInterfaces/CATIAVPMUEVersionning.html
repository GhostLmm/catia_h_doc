#ifndef CATIAVPMUEVersionning_h #define CATIAVPMUEVersionning_h // COPYRIGHT DASSAULT SYSTEMES 2000 /** * @CAA2Level L1 * @CAA2Usage U5 */ #ifdef _WINDOWS_SOURCE #ifdef __VPMProIDL #define ExportedByVPMProIDL __declspec(dllexport) #else #define ExportedByVPMProIDL __declspec(dllimport) #endif #else #define ExportedByVPMProIDL #endif #include "CATBSTR.h" #include "CATIABase.h" #include "CATUnicodeString.h" class CATIAVPMObjectVersion;
