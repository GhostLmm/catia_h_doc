Any other value counts for negative sign. */ virtual int GetSign() const = 0; /** * Set sign of a Signed Spec. * @param i * Sign to set. * <br><b>Legal values</b>: 1 counts for positive sign. Any other value counts for negative sign. */ virtual void SetSign(const int i) = 0; /** * Get Simple Specification composing Signed Specification. * @param oSpec * The simple Specification retrieved. */ virtual void GetSpec (CATICfgSimpleSpecification_var &oSpec) const = 0;
