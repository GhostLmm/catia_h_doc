#ifndef ENOVIDocumentable_H #define ENOVIDocumentable_H // COPYRIGHT DASSAULT SYSTEMES 2000 /** * @CAA2Level L1 * @CAA2Usage U3 */ // ================================================================= // Includes // ================================================================= // System #include "GUIDVPMInterfaces.h" #include "CATBaseUnknown.h" #include "CATUnicodeString.h" #include "CATListOfCATIAVPMDocumentVersion.h" // ================================================================= // Forward classes declaration // ================================================================= class CATICfgFilter;
