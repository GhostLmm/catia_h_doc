public: /** * Gets the reference plane. * @param oPlane * reference plane */ virtual HRESULT GetReferencePlane(CATIMmiMechanicalFeature_var& oPlane) = 0; /** * Sets the reference plane. * @param iPlane * reference plane */ virtual HRESULT SetReferencePlane(const CATIMmiMechanicalFeature_var iPlane) = 0;
