#ifndef CATIMSHOperatorManager_h #define CATIMSHOperatorManager_h // COPYRIGHT DASSAULT SYSTEMES 2000 /** * @CAA2Level L0 * @CAA2Usage U3 */ #include "MSHModel.h" #include "CATBaseUnknown.h" #include "CATLISTV_CATISpecObject.h" class CATISpecObject; extern ExportedByMSHModel IID IID_CATIMSHOperatorManager; /** * Interface used to manage <b>Mesh Operators</b> within a <b>Mesh Part</b>. * <b>Role</b>: A <b>Mesh Operator</b> is an associative specification which is used * to modify the mesh generated within a <b>Mesh Part</b> @href CATIMSHMeshPart. * <p>At the end of mesh generation within a <b>Mesh Part</b>, the different <b>Mesh Operators</b> * are called to in order to modify the mesh, see @href CATIMSHOperator. */ class ExportedByMSHModel CATIMSHOperatorManager: public CATBaseUnknown { CATDeclareInterface; public: /** * Add a new operator. * <br><b>Role</b>: This method is called to create a new <b>Mesh Operator</b> within * the <b>Mesh Part</b>.
