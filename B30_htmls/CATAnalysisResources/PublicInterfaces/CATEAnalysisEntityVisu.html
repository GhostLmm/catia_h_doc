class CATPathElement; class CATRepPath; //----------------------------------------------------------------------- /** * Adapter for the implementation of CATISamEntityVisu interface. * This adapter allow a BOA implementation. * @see CATISamEntityVisu */ class ExportedByCATAnalysisResources CATEAnalysisEntityVisu: public CATISamEntityVisu { public: // Standard constructors and destructors for an implementation class // ----------------------------------------------------------------- CATEAnalysisEntityVisu ();
