#ifndef CATKinDefs_h #define CATKinDefs_h // // COPYRIGHT DASSAULT SYSTEMES 2000 //=================================================================== // // CATIKinMechanism.h // This header defines the basic types of kinematic joints and commands. // //=================================================================== // // Organization : dg/rddfl/mps/mam // //=================================================================== /** * @CAA2Level L1 * @CAA2Usage U1 */ #include "KinematicsItf.h" /** * * Definition of kinematic joints types * */ /** * @param CATKinPrismaticJoint * The joint is prismatic. */ ExportedByKinematicsItf extern const char* CATKinPrismaticJoint;
