#ifndef CATEwrWireGroup_H #define CATEwrWireGroup_H // COPYRIGHT DASSAULT SYSTEMES 2004 /** * @CAA2Level L1 * @CAA2Usage U1 */ // module declaration #include "ExportedByCATEwrInterfaces.h" // for class variable declaration #include "CATUnicodeString.h" #include "CATListOfCATUnicodeString.h" /** * Class dedicated to wire group. * <b>Role</b>: This class is used to represent wire group and its attributes */ class ExportedByCATEwrInterfaces CATEwrWireGroup { public: /** * The identifier of the WireGroup. * <br><b>Mandatory</b> */ CATUnicodeString Id;
