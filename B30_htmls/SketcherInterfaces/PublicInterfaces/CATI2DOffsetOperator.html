#else extern "C" const IID IID_CATI2DOffsetOperator; #endif class ExportedByCATSketcherInterfaces CATI2DOffsetOperator : public CATI2DOffset { CATDeclareInterface; public: /** * Retrieves the propagation type. * @param oPropagType * [out] The propagation type. * <br>e.g.: <code>OffsetPropagType_Point</code> * @return <tt>HRESULT</tt> */ virtual HRESULT GetPropagationType(int *oPropagType) = 0;
