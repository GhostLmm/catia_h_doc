#endif class CATString; class CATPrinter; class CATPrintStream; class CATPrintFileDevice; /** * Class to create a printer device. * <br> * <b>Role</b>: This class allows to create a device based on a printer, or for printing to a file * instead sending data to the printer. * * @see CATPrinter, CATPrintFileDevice */ class ExportedByCATPrintBase CATPrinterDevice : public CATPrintFileDevice { CATDeclareClass;
