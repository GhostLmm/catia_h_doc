#else extern "C" const IID IID_CATIDftCustomFormat ; #endif // System #include "CATBooleanDef.h" #include "CATUnicodeString.h" /** * <B>Role:</B> CATIDftCustomFormat is an interface dedicated to change the generic informations carried by a format that are * consequently shared among sheets associated with this format. * @see CATIDftFormat. */ class ExportedByDraftingItfCPP CATIDftCustomFormat: public CATBaseUnknown { CATDeclareInterface;
