#else extern "C" const IID IID_CATIProdDraftingProperties; #endif /** * Interface to manage the drafting properties of a product. * <br><b>Role</b>: Allows to specify the properties that should * be used when generating views in the Drafting workbench. */ class ExportedByCATAssemblyInterfaces CATIProdDraftingProperties : public CATBaseUnknown { CATDeclareInterface;
