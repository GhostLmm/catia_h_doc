#ifndef CATEdge_H #define CATEdge_H // COPYRIGHT DASSAULT SYSTEMES 1999 /** * @CAA2Level L1 * @CAA2Usage U3 */ #include "CATGMModelInterfaces.h" #include "CATGeoDef.h" #include "CATCell.h" #include "ListVOfLong.h" class CATBody; class CATDomain; class CATShell; class CATLoop; class CATBoundaryIterator; class CATBoundedCellsIterator; class CATFace; class CATVertex; class CATEdgeCurve; class CATPCurve; class CATPointOnEdgeCurve; class CATCrvParam; class CATMathVector; class CATSurLimits; #include "CATTopData.h" #ifndef LOCAL_DEFINITION_FOR_IID extern ExportedByCATGMModelInterfaces IID IID_CATEdge ; #else extern "C" const IID IID_CATEdge ; #endif /** * Interface representing a topological edge. * <br> An edge is a topological cell which geometry is an edge-curve.
