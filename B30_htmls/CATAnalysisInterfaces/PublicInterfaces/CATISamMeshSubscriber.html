#ifndef CATISamMeshSubscriber_H #define CATISamMeshSubscriber_H // COPYRIGHT DASSAULT SYSTEMES 2000 //=================================================================== /** * @CAA2Level L1 * @CAA2Usage U4 CATEAnalysisEntityPreproSubscriber */ //=================================================================== #include "CATAnalysisInterface.h" #include "CATBaseUnknown.h" #include "CATBoolean.h" #ifndef LOCAL_DEFINITION_FOR_IID extern ExportedByCATAnalysisInterface IID IID_CATISamMeshSubscriber ; #else extern "C" const IID IID_CATISamMeshSubscriber ; #endif //------------------------------------------------------------------ /** * Management of the entities notified by mesh modification. * <b>Role:</b> Interface designed to manage <b>Analysis Entities</b> * notified by mesh modification.<br> * <p> * In the Analysis Model, when <b>the mesh</b> is modified * it is necessary to notify some entities of this changes.
