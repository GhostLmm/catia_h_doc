//=================================================================== // COPYRIGHT DASSAULT SYSTEMES 2000 //=================================================================== # ifndef CATDimDefinition_H # define CATDimDefinition_H /** * @CAA2Level L1 * @CAA2Usage U1 */ #include"CATDrwDimEnum.h" #include"CATBooleanDef.h" #include "CATErrorDef.h" #include"DraftingItfCPP.h" /** * Class to initialize basic informations necessary to the dimension creation. */ class ExportedByDraftingItfCPP CATDimDefinition { public: // constructor (easier initializations) CATDimDefinition();
