// COPYRIGHT DASSAULT SYSTEMES 1994-2000 //=================================================================== // // CAAEOmmFactory.h // Provides implementation to interface CAAIAOmmFactory // //=================================================================== #ifndef CAAEOmmFactory_H #define CAAEOmmFactory_H // System framework #include "CATBaseUnknown.h" // needed as base implementation class #include "CATBSTR.h" // needed to use OLE strings class CATBaseDispatch;
