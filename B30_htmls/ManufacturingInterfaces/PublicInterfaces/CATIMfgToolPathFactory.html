public: /** * Creates a compound traject inside the tool path container. * @return * The compound traject object **/ virtual CATIMfgCompoundTraject_var CreateMfgCompoundTraject () = 0 ; /** * Creates a tool path cycle inside the tool path container. * @param iUserSyntax * The list of strings describing the syntax of cycle * @return * The tool path cycle object **/ virtual CATIMfgToolPathCycle_var CreateMfgToolPathCycle (const CATListValCATUnicodeString& iUserSyntax) = 0;
