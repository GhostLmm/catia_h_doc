#else extern "C" const IID IID_CATIMfgProgramSimulationTools ; #endif //------------------------------------------------------------------ /** * Interface implemented on the Manufacturing Program for ISO based simulation. * The rule of this interface is to give informations on the tools (Cutter Compensation) and * Workpiece Origins */ class ExportedByCATMfgSimulationItfCPP CATIMfgProgramSimulationTools: public CATBaseUnknown { CATDeclareInterface;
